Étape 1 — Créer votre base de données gratuite
Supabase est gratuit (jusqu'à 500 MB, largement suffisant pour des années de données). Suivez ces étapes :

1. Allez sur supabase.com → cliquez Start for free
2. Créez un compte (Google ou email)
3. Cliquez New project
4. Choisissez un nom (ex: mgt-crm), une région proche (Europe West), un mot de passe fort
5. Attendez ~2 minutes que le projet se crée
6. Passez à l'étape suivante
Étape 2 — Créer les tables
Dans Supabase, allez dans SQL Editor (menu gauche), cliquez New query, copiez-collez ce SQL et cliquez Run :
-- Tables MGT CRM CREATE TABLE IF NOT EXISTS mgt_clients ( id TEXT PRIMARY KEY, name TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS mgt_formats ( id TEXT PRIMARY KEY, dim TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS mgt_types ( id TEXT PRIMARY KEY, name TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS mgt_users ( id TEXT PRIMARY KEY, login TEXT UNIQUE NOT NULL, pass TEXT NOT NULL, role TEXT DEFAULT 'operator', created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS mgt_prices ( client_id TEXT PRIMARY KEY, price NUMERIC DEFAULT 0 ); CREATE TABLE IF NOT EXISTS mgt_entries ( id TEXT PRIMARY KEY, ts TIMESTAMPTZ NOT NULL, client_id TEXT, format_id TEXT, type_id TEXT, client_name TEXT, format_dim TEXT, type_name TEXT, qty INTEGER DEFAULT 0, waste INTEGER DEFAULT 0, operator TEXT, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Accès public (RLS désactivé pour usage interne) ALTER TABLE mgt_clients ENABLE ROW LEVEL SECURITY; ALTER TABLE mgt_formats ENABLE ROW LEVEL SECURITY; ALTER TABLE mgt_types ENABLE ROW LEVEL SECURITY; ALTER TABLE mgt_users ENABLE ROW LEVEL SECURITY; ALTER TABLE mgt_prices ENABLE ROW LEVEL SECURITY; ALTER TABLE mgt_entries ENABLE ROW LEVEL SECURITY; CREATE POLICY "allow_all" ON mgt_clients FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all" ON mgt_formats FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all" ON mgt_types FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all" ON mgt_users FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all" ON mgt_prices FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all" ON mgt_entries FOR ALL USING (true) WITH CHECK (true);
